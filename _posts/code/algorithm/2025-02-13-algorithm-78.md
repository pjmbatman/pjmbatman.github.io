def solution(s):
    
    # works for first letter too
    prev = " "
    result = ""
    for letter in s:
        if letter == " ":
            result += letter
            prev = letter
            continue
            
        if prev == " ":
            result += letter.upper()
            prev = None
            
        else:
            result += letter.lower()
            prev = letter
            
    return result

def solution(s):
    
    result = ""
    upper_trigger = True
    for letter in s:
        if letter == " ":
            upper_trigger = True
        else:
            if upper_trigger:
                letter = letter.upper()
                upper_trigger = False
            else:
                letter = letter.lower()
        result += letter
        
    return result
