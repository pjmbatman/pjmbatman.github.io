def solution(diffs, times, limit):
    left = 1
    right = max(diffs)
    candidates = []
    while left <= right:
        level = (left + right) // 2
        time_prev = 0
        total_time = 0
        for diff, time_cur in zip(diffs, times):
            if diff <= level:
                time_spent = time_cur
            else:
                time_spent = (diff-level) * (time_cur + time_prev) + time_cur
            time_prev = time_cur
            total_time += time_spent
        
        if total_time <= limit:
            right = level - 1
            candidates.append(level)
        else:
            left = level + 1
            
    
    return min(candidates)
