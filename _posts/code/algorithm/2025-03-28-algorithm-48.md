def is_palindrome(string):
    size = len(string)
    if size % 2 == 0:
        if string[:size//2] == string[size//2:][::-1]:
            return True
        return False
    else:
        if string[:size//2] == string[size//2+1:][::-1]:
            return True
        return False
            

def solution(s):
    dist = len(s)
    while dist > 0:
        for i in range(0, len(s)-(dist-1)):
            if is_palindrome(s[i:i+dist]):
                return len(s[i:i+dist])
                
        dist -= 1
