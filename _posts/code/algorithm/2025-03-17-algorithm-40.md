import math

def solution(r1, r2):
    
    def big_in(x, y, r2):
        if x**2 + y**2 <= r2**2:
            return True
        else:
            return False
        
    def small_out(x, y, r1):
        if x**2 + y**2 >= r1**2:
            return True
        else:
            return False
    
    count = 0
    
    for i in range(0, r2+1):
        for j in range(0, r2+1):
            if big_in(i, j, r2) and small_out(i, j, r1):
                count += 1

    return count * 4 - 8
