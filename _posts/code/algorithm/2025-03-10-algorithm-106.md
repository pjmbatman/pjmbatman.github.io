def solution(s):
    
    bin_count = 0
    zero_count = 0
    
    while len(s) != 1 and s != "1":
        prev_len = len(s)
        new_s = s.replace("0", "")
        next_len = len(new_s)
        zero_count += prev_len - next_len
        
        s = bin(next_len)[2:]
        bin_count += 1
    
    return [bin_count, zero_count]
