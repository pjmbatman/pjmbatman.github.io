```python
    def dfs(index, result):

            if result == target:
                return 1
            return 0
        
        num = numbers[index]
        
        plus_count = dfs(index+1, result+num)
```

```python

from collections import deque

def solution(numbers, target):
    
    queue = deque([(0, 0)])
    count = 0
    while queue:
        idx, total = queue.popleft()
        if idx == len(numbers):
            if total == target:
                count += 1
            continue
        queue.append((idx+1, total+numbers[idx]))
        queue.append((idx+1, total-numbers[idx]))
        
    return count
```
