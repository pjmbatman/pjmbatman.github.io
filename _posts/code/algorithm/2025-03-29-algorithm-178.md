import heapq
def solution(n, works):
    
    heap = [-x for x in works]
    heapq.heapify(heap)
    
    while n > 0:
        largest = heapq.heappop(heap)
        if largest == 0:
            break
        heapq.heappush(heap, largest + 1)
        n -= 1
    
    total = 0
    for i in range(len(heap)):
        total += heap[i]**2
        
    return total
