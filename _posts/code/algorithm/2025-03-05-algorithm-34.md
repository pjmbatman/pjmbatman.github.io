import heapq
def solution(scoville, K):
    
    heapq.heapify(scoville)
    count = 0
    if scoville[0] >= K:
        return count
    
    while len(scoville) > 1:
        least = heapq.heappop(scoville)

        mixed = least + second*2
        heapq.heappush(scoville, mixed)
        count += 1
        if scoville[0] >= K:
            return count
        
    return -1
