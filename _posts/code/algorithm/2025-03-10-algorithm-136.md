def solution(orders, course):
    
    result = []
    
    for comb_len in course:
        combs = []

        for order in orders:
            combs.extend(combinations(sorted(order), comb_len))
        
        order_count = Counter(combs)
        
        
        if order_count:
            max_order = max(order_count.values())

            if max_order >= 2:

                    if value == max_order:

                        result.append(''.join(key))
            
    return sorted(result)
