import math
def solution(n, stations, w):
    
    current = 1
    count = 0
    stations.sort()
    for station in stations:
        end = station - w
        if end > current:
            gap = end - current 
            count += math.ceil(gap / (2*w + 1))
        current = station + w + 1
        
    if current <= n:
        gap = n+1 - current
        count += math.ceil(gap / (2*w + 1))
        
    return count
