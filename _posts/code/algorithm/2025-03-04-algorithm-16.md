from collections import deque

def solution(m, n, board):
    board = [list(row) for row in board]
    # ì œê±° ëŒ€ìƒ í™•ì¸

    count = 0
    
    while True:

            for j in range(n):

                    continue

                    erase_map[i][j] = 0
                    erase_map[i][j-1] = 0
                    erase_map[i-1][j-1] = 0
                    erase_map[i-1][j] = 0

        for i in range(m):

        for j in range(n):

                    queue.append((i,j))
                    flag = True
                elif board[i][j] != '0' and flag:

                    board[row][col] = board[i][j]
