def solution(N, stages):
    
    # stage : fail ratio
    stage_dict = {}
    users = len(stages)
    
    for stage in range(1, N + 1):
        if users > 0:
            stage_dict[stage] = stages.count(stage) / users
            
        else:
            stage_dict[stage] = 0
            
        users -= stages.count(stage)
            
    result = sorted(stage_dict, key=lambda x: stage_dict[x], reverse = True)

    return result
