def solution(numbers):

    result = []
    
    for i in range(len(numbers)):
        is_found = False
        for j in range(i, len(numbers)):
            current = numbers[i]
            compare = numbers[j]
            if current < compare:
                result.append(compare)
                is_found = True
                break
        if not is_found:
            result.append(-1)

    return result
