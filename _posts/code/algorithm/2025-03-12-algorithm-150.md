from collections import deque

def solution(x, y, n):
    
    queue = deque([(x, 0)])
    visited = set()
    
    while queue:
        x, count = queue.popleft()
        if x == y:
            return count
        
        processes = [(x + n), (x * 2), (x * 3)]
        
        count += 1
        
        for process in processes:
            if process <= y and process not in visited:
                visited.add(process)
                queue.append((process, count))
        
    return -1
