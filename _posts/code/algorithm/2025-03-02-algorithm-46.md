"""

from collections import deque

def solution(maps):
    
    # (row, col, dist)
    queue = deque([(0,0,1)])
    directions = [(-1,0), (1,0), (0,-1), (0,1)]
    m = len(maps)
    n = len(maps[0])
    
    is_visted = [[False for _ in range(n)] for _ in range(m)]
    
    while queue:
        row, col, dist = queue.popleft()
        
        if row == m-1 and col == n-1:
            return dist
        
        for (x,y) in directions:
            new_row = row+x
            new_col = col+y
            if 0 <= new_row < m and 0 <= new_col < n and maps[new_row][new_col] == 1 and is_visted[new_row][new_col] == False :
                is_visted[new_row][new_col] = True
                queue.append((new_row, new_col, dist+1))
                
    return -1
