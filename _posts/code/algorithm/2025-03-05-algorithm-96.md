from collections import deque

def solution(prices):
    
    temp = prices.copy()
    queue = deque(temp)
    
    current_index = 0
    result = []
    
    while queue:
        current_price = queue.popleft()
        time = 0
        is_drop = False
        for index in range(current_index + 1, len(prices)):
            time += 1
            if prices[index] < current_price:
                result.append(time)
                is_drop = True
                break

        current_index += 1
        if not is_drop:
            result.append(time)
            

    return result
