def solution(n, info):
    
    result = []
    
    def dfs(lion, index, arrow_left):
        if index == len(info):
            if arrow_left:
                lion[-1] += arrow_left
            diff = 0
            for score in range(len(info)):
                if lion[score] > info[score]:
                    diff += (10 - score)
                elif lion[score] < info[score]:
                    diff -= (10 - score)
                else:
                    if lion[score] != 0:
                        diff -= (10 - score)
                        
            result.append([diff, lion])
            return
        
        a = lion.copy()
        b = lion.copy()
        
        if info[index] + 1 <= arrow_left:
            a.append(info[index] + 1)
            dfs(a, index + 1, arrow_left - (info[index] + 1))
        
        b.append(0)
        dfs(b, index + 1, arrow_left)
        
    dfs([], 0, n)
    
    return max(result, key=lambda x:(x[0], tuple(x[1])[::-1]))[1] if max(result, key=lambda x:x[0])[0] > 0 else [-1]
